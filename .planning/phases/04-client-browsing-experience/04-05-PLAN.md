Create `client/src/api/content.ts` with functions that call the browse API:

```typescript
import { api } from "./client";
import type { BrowsePageData, LiveTvData, ContentDetail, Content, SearchResults } from "@/types/content";

export async function fetchHomePageData(): Promise<BrowsePageData> {
  const { data } = await api.get("/browse");
  return data;
}

export async function fetchBrowsePageData(type: "movies" | "series" | "documentaries"): Promise<BrowsePageData> {
  const { data } = await api.get(`/browse/${type}`);
  return data;
}

export async function fetchLiveTvData(): Promise<LiveTvData> {
  const { data } = await api.get("/browse/live-tv");
  return data;
}

export async function fetchTitleDetail(id: string): Promise<ContentDetail> {
  const { data } = await api.get(`/browse/title/${id}`);
  return data;
}

export async function fetchSimilarTitles(id: string): Promise<Content[]> {
  const { data } = await api.get(`/browse/title/${id}/similar`);
  return data;
}

export async function searchContent(query: string): Promise<SearchResults> {
  const { data } = await api.get("/browse/search", { params: { q: query } });
  return data;
}
```

These are plain async functions (not hooks). They'll be used inside TanStack Query's `queryFn`.
