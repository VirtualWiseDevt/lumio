```typescript
import cron from "node-cron";
import { prisma } from "../config/database.js";
import { sendEmail, buildPreExpiryEmail, buildPostExpiryEmail } from "../services/email.service.js";

export function startSubscriptionExpiryJob(): void {
  // Run every hour at minute 30 (offset from session cleanup at :00)
  cron.schedule("30 * * * *", async () => { ... });
  console.log("[EXPIRY_JOB] Scheduled hourly subscription expiry checks");
}
```

The cron callback logic:

1. **Pre-expiry 2 days** -- Find ACTIVE subscriptions where:
   - `status: "ACTIVE"`
   - `expiresAt > now AND expiresAt <= now + 2 days`
   - `notifiedPreExpiry2Day: null` (not yet notified)
   - Include `user` and `plan` relations

   For each result:
   - Build email with `buildPreExpiryEmail(sub.user.name, { planName: sub.plan.name, expiresAt: sub.expiresAt, daysRemaining: 2 })`
   - Fire-and-forget: `sendEmail(...).catch(err => console.error(...))`
   - Mark as notified: `prisma.subscription.update({ where: { id: sub.id }, data: { notifiedPreExpiry2Day: now } })`
   - Log: `[EXPIRY_JOB] Sent 2-day pre-expiry to ${sub.user.email}`

2. **Pre-expiry 1 day** -- Find ACTIVE subscriptions where:
   - `status: "ACTIVE"`
   - `expiresAt > now AND expiresAt <= now + 1 day`
   - `notifiedPreExpiry1Day: null` (not yet notified for 1-day)
   - Include `user` and `plan` relations

   For each result:
   - Build email with `buildPreExpiryEmail(sub.user.name, { planName: sub.plan.name, expiresAt: sub.expiresAt, daysRemaining: 1 })`
   - Fire-and-forget send
   - Mark: `prisma.subscription.update({ data: { notifiedPreExpiry1Day: now } })`
   - Log count

3. **Post-expiry 1 day** -- Find subscriptions where:
   - `status: "ACTIVE"` (status hasn't been changed yet, but expiresAt has passed)
   - `expiresAt < now` (already expired)
   - `expiresAt >= now - 2 days` (expired within last 2 days -- don't notify ancient ones)
   - `notifiedPostExpiry: null`
   - Include `user` and `plan` relations

   For each result:
   - FIRST check if the user has ANY other active subscription with `expiresAt > now`. If yes, SKIP (they renewed).
   - Build email with `buildPostExpiryEmail(sub.user.name, { planName: sub.plan.name, expiredAt: sub.expiresAt })`
   - Fire-and-forget send
   - Mark: `prisma.subscription.update({ data: { notifiedPostExpiry: now } })`
   - Log count

**Key implementation details:**
- Wrap entire callback in try/catch (matches existing job patterns)
- Process each subscription in a for-of loop (not Promise.all -- avoid hammering SMTP)
- Log summary at end: `[EXPIRY_JOB] Processed: X 2-day, Y 1-day pre-expiry, Z post-expiry`
- Only log if any notifications were sent (quiet runs produce no output, matching session cleanup pattern)
- The `notifiedPreExpiry2Day` field is set even if the email send fails (to prevent retry storms). The email is fire-and-forget anyway.
- For post-expiry: check for renewal by querying if user has a DIFFERENT active subscription. Use `prisma.subscription.findFirst({ where: { userId, status: "ACTIVE", expiresAt: { gt: now }, id: { not: sub.id } } })`.

**IMPORTANT:**
- Use .js extensions on all imports
- The cron expression `"30 * * * *"` runs at minute 30 of every hour (offset from session cleanup at :00 and reconciliation at */2)
- All email sends are fire-and-forget with .catch(log)
